# Linux-specific dependencies for system tray support

# Required for pystray to work on Linux with GTK backend
PyGObject>=3.42.0

# Cairo bindings for Python
# Required for rendering operations
pycairo>=1.20.0

# For Fedora:
#   sudo dnf install python3-gobject gtk3 libappindicator-gtk3
